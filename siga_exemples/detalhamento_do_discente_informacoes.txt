		window.acessibilidade = true;

			jQuery(this).addClass('acessibilidade');
		});

			iFrame.contentDocument.getElementsByTagName('body')[0].className = 'acessibilidade';
		}
		else if ( iFrame.contentWindow ) 
		{ // IE
			iFrame.contentWindow.getElementsByTagName('body')[0].className = 'acessibilidade';
		}
		
		for(var i = 0; i < window.frames.length;i++){
			if(window.frames[i] && window.frames[i].altocontraste){
				window.frames[i].altocontraste();
			}
		}		
	}
	
	function removeAltocontraste(){
		window.acessibilidade = false;

				window.frames[i].removeAltocontraste();
			}
		}
	}
